class Solution {
public:
     void Rotateta(vector<int>&nums , int st , int end ){
        while(st<end){
            int temp = nums[st];
            nums[st]= nums[end];
            nums[end] =temp;
            st++;
            end--;
        }
     } 
    bool check(vector<int>& nums) {
        int i = 0 ;
        int j = i+1 ;
        while (i<nums.size()){
            if (nums[i]<nums[j]){
                i++;
                j++;
            }
            else {
                break ;
            }

        }
        int idx = j-1 ;
        Rotateta(nums , 0 , i-1);
        Rotateta(nums , idx , nums.size()-1);
        Rotateta(nums , 0 ,nums.size()-1 );

    
        

 //////// TYPES OF COMPLEXITY /////////////////
 1> Best case  == big O
 2> Averaage case == omega 
 3> Worst case == theta 


        
         
        
    }
};
